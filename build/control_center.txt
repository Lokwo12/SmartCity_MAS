:- dynamic request/3, tried/2, truck_state/2, started/0, cycle_counter/1.
:- dynamic req_seq/1, inflight/6, processed_reply/3, completed_req/1, dead_letter/4.
:- dynamic last_cycle_ts/1, pending_dup_notified/1.
:- dynamic rr_idx/1.

truck(truck1).
truck(truck2).
truck(truck3).

bin(smart_bin1).
bin(smart_bin2).
bin(smart_bin3).

shared_token(city_token_2026).

deltaT(3).
reply_ttl(6).
assign_ack_ttl(6).
completion_ttl(10).
reset_ack_ttl(3).
cycle_interval_ms(1000).

startI :>
    retractall(started),
    assert(started),
    deltaT(D),
    retractall(cycle_counter(_)),
    assert(cycle_counter(D)),
    retractall(req_seq(_)),
    assert(req_seq(1)),
    retractall(rr_idx(_)),
    assert(rr_idx(1)),
    retractall(last_cycle_ts(_)),
    statistics(walltime,[Now,_]),
    assert(last_cycle_ts(Now)),
    init_trucksA,
    true.

init_trucksA :-
    retractall(truck_state(_,_)),
    forall(truck(T), assert(truck_state(T,idle))).

cycleI :>
    monitor_systemI.

monitor_systemI :>
    check_stuck_requestsA,
    process_timeoutsA.

check_stuck_requestsA :- true.

new_req_idA(ReqId) :-
    req_seq(N),
    ReqId is N,
    N1 is N + 1,
    retract(req_seq(N)),
    assert(req_seq(N1)).

pending_bin(Bin) :-
    request(_,Bin,_).

trusted_bin(B)   :- bin(B).
trusted_truck(T) :- truck(T).

short_bin_id(Bin,Short) :-
    atom_concat(smart_bin,N,Bin),
    !,
    atom_concat(sb,N,Short).
short_bin_id(Bin,Bin).

short_truck_id(Truck,Short) :-
    atom_concat(truck,N,Truck),
    !,
    atom_concat(t,N,Short).
short_truck_id(Truck,Truck).

telemetryA(Level,Name,ReqId,Bin,Truck,Note) :-
    messageA(logger,
        send_message(log(Level,event(Name,req(ReqId),bin(Bin),truck(Truck),note(Note)),control_center),control_center),
        control_center).

persist_dead_letterA(ReqId,Bin,Reason) :-
    statistics(walltime,[Ts,_]),
    assert(dead_letter(ReqId,Bin,Reason,Ts)),
    telemetryA(error,dead_letter,ReqId,Bin,none,Reason).

bin_fullE(Bin) :>
    trusted_bin(Bin),
    \+ pending_bin(Bin),
    !,
    retractall(pending_dup_notified(Bin)),
    format('[CONTROL_CENTER] BIN_FULL received from bin=~w~n',[Bin]),
    new_req_idA(ReqId),
    clear_tried_listA(Bin),
    telemetryA(info,request_opened,ReqId,Bin,none,accepted),
    select_truckI(ReqId,Bin).

bin_fullE(Bin) :>
    pending_bin(Bin),
    \+ pending_dup_notified(Bin),
    !,
    assert(pending_dup_notified(Bin)),
    telemetryA(warn,duplicate_bin_full,0,Bin,none,ignored).

bin_fullE(Bin) :>
    pending_bin(Bin),
    pending_dup_notified(Bin),
    !,
    true.

    shared_token(Token), trusted_bin(From),
    fire_event(bin_full(Bin)).

send_message(bin_full(Bin),From) :-

    fire_event(bin_full(Bin)).

send_message(bin_full(Bin,Token),From) :-
    shared_token(Token), trusted_bin(From),
    fire_event(bin_full(Bin)).

send_message(agree(pickup(Bin),T),From) :-

send_message(job_accept(T,Bin),From) :-
    fire_event(job_accept(T,Bin,0)).

send_message(job_accept(T,Bin,ReqId,Token),From) :-
    shared_token(Token),
    fire_event(job_accept(T,Bin,ReqId)).

send_message(refuse(pickup(Bin),T),From) :-
    fire_event(job_refuse(T,Bin,0)).

send_message(job_refuse(T,Bin),From) :-

send_message(job_refuse(T,Bin,ReqId,Token),From) :-
    shared_token(Token),
    fire_event(job_refuse(T,Bin,ReqId)).

send_message(assignment_ack(T,Bin,ReqId,Token),From) :-

    fire_event(assignment_ack(T,Bin,ReqId)).

send_message(assignment_ack(T,Bin,ReqId),From) :-
    fire_event(assignment_ack(T,Bin,ReqId)).

send_message(inform(collection_complete(Bin),_),From) :-
    fire_event(collection_complete(Bin,0)).

send_message(collection_complete(Bin),From) :-
    fire_event(collection_complete(Bin,0)).

send_message(collection_complete(Bin,ReqId,Token),From) :-
    shared_token(Token),

send_message(collection_complete(Bin,ReqId),From) :-
    fire_event(collection_complete(Bin,ReqId)).

send_message(inform(collection_failed(T,Bin),_),From) :-
    fire_event(collection_failed(T,Bin,0)).

    fire_event(collection_failed(T,Bin,0)).

send_message(collection_failed(T,Bin,ReqId,Token),From) :-

    fire_event(collection_failed(T,Bin,ReqId)).

    fire_event(collection_failed(T,Bin,ReqId)).

send_message(reset_ack(Bin,ReqId,Token),From) :-
    shared_token(Token), trusted_bin(From),
    fire_event(reset_ack(Bin,ReqId)).

fire_event(E) :-
    ( catch(call(eve(E)),_,fail)
    ; catch(call(evi(E)),_,fail)
    ), !.
fire_event(_).

select_truckI(ReqId,Bin) :>
    findall(T,
        (truck(T), truck_state(T,idle), \+ tried(Bin,T)),
        IdleCandidates),
    IdleCandidates \= [],
    !,
    dispatchI(ReqId,Bin,IdleCandidates).

select_truckI(ReqId,Bin) :>
    findall(T,
        (truck(T), \+ tried(Bin,T)),
        Candidates),
    dispatchI(ReqId,Bin,Candidates).

choose_candidateA(Candidates,T) :-
    rr_idx(I),
    length(Candidates,N),
    N > 0,
    Pos is ((I-1) mod N) + 1,
    nth1(Pos,Candidates,T),
    I1 is I + 1,
    retract(rr_idx(I)),
    assert(rr_idx(I1)).

dispatchI(ReqId,Bin,[]) :>
    retractall(inflight(ReqId,Bin,_,_,_,_)),
    assert(inflight(ReqId,Bin,none,awaiting_retry,2,0)),
    telemetryA(warn,no_truck_available_retry,ReqId,Bin,none,awaiting_retry).

dispatchI(ReqId,Bin,Candidates) :>
    choose_candidateA(Candidates,T),
    assert(request(ReqId,Bin,T)),
    reply_ttl(Ttl),
    retractall(inflight(ReqId,Bin,T,_,_,_)),
    assert(inflight(ReqId,Bin,T,awaiting_reply,Ttl,0)),
    format('[CONTROL_CENTER] DISPATCH request=~w bin=~w truck=~w~n',[ReqId,Bin,T]),
    shared_token(Token),
    messageA(T, send_message(pickup_request(Bin,ReqId,Token),control_center), control_center),
    telemetryA(info,pickup_request_sent,ReqId,Bin,T,awaiting_reply).

job_acceptE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    inflight(ReqId,Bin,T,awaiting_reply,_,_),
    \+ processed_reply(ReqId,T,accept),
    !,
    format('[CONTROL_CENTER] ACCEPT request=~w bin=~w truck=~w~n',[ReqId,Bin,T]),
    assert(processed_reply(ReqId,T,accept)),
    retractall(truck_state(T,_)),
    assert(truck_state(T,busy)),
    completion_ttl(Ttl),
    retractall(inflight(ReqId,Bin,T,_,_,_)),
    assert(inflight(ReqId,Bin,T,awaiting_completion,Ttl,0)),
    telemetryA(info,pickup_accepted_start_collection,ReqId,Bin,T,awaiting_completion).

job_refuseE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    inflight(ReqId,Bin,T,awaiting_reply,_,_),
    \+ processed_reply(ReqId,T,refuse),
    !,
    format('[CONTROL_CENTER] REFUSE request=~w bin=~w truck=~w -> reselection~n',[ReqId,Bin,T]),
    assert(processed_reply(ReqId,T,refuse)),
    retract(request(ReqId,Bin,T)),
    retractall(inflight(ReqId,Bin,T,_,_,_)),
    assert(tried(Bin,T)),
    telemetryA(warn,truck_refused,ReqId,Bin,T,retry_select),
    select_truckI(ReqId,Bin).

job_acceptE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    \+ inflight(ReqId,Bin,T,awaiting_reply,_,_),
    telemetryA(info,late_accept_ignored,ReqId,Bin,T,non_reply_stage).

job_refuseE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    \+ inflight(ReqId,Bin,T,awaiting_reply,_,_),
    telemetryA(info,late_refuse_ignored,ReqId,Bin,T,non_reply_stage).

assignment_ackE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    inflight(ReqId,Bin,T,awaiting_assign_ack,_,R),
    !,
    format('[CONTROL_CENTER] ASSIGNMENT_ACK request=~w bin=~w truck=~w~n',[ReqId,Bin,T]),
    completion_ttl(Ttl),
    retract(inflight(ReqId,Bin,T,awaiting_assign_ack,_,R)),
    assert(inflight(ReqId,Bin,T,awaiting_completion,Ttl,0)),
    telemetryA(info,assignment_acked,ReqId,Bin,T,awaiting_completion).

collection_completeE(Bin,ReqId) :>
    request(ReqId,Bin,T),
    \+ completed_req(ReqId),
    !,
    format('[CONTROL_CENTER] COLLECTION_COMPLETE request=~w bin=~w truck=~w~n',[ReqId,Bin,T]),
    assert(completed_req(ReqId)),
    retractall(truck_state(T,_)),
    assert(truck_state(T,idle)),
    retractall(inflight(ReqId,Bin,T,_,_,_)),
    shared_token(Token),
    messageA(Bin, send_message(reset_bin(Bin,ReqId,Token),control_center), control_center),
    retractall(request(ReqId,Bin,_)),
    retractall(pending_dup_notified(Bin)),
    clear_tried_listA(Bin),
    messageA(logger, send_message(log(info,collection_complete,Bin),control_center), control_center),
    telemetryA(info,reset_sent_async,ReqId,Bin,T,closed_after_reset_send).

collection_completeE(Bin,0) :>
    request(ReqId,Bin,_),
    ReqId > 0,
    fire_event(collection_complete(Bin,ReqId)).

collection_failedE(T,Bin,ReqId) :>
    request(ReqId,Bin,T),
    !,
    format('[CONTROL_CENTER] COLLECTION_FAILED request=~w bin=~w truck=~w -> reselection~n',[ReqId,Bin,T]),
    retractall(truck_state(T,_)),
    assert(truck_state(T,idle)),
    retract(request(ReqId,Bin,T)),
    retractall(inflight(ReqId,Bin,T,_,_,_)),
    assert(tried(Bin,T)),
    telemetryA(warn,collection_failed,ReqId,Bin,T,retry_select),
    select_truckI(ReqId,Bin).

collection_failedE(T,Bin,0) :>
    request(ReqId,Bin,T),
    ReqId > 0,
    fire_event(collection_failed(T,Bin,ReqId)).

reset_ackE(Bin,ReqId) :>
    inflight(ReqId,Bin,T,awaiting_reset_ack,_,_),
    !,
    format('[CONTROL_CENTER] RESET_ACK request=~w bin=~w truck=~w -> closed~n',[ReqId,Bin,T]),
    retractall(inflight(ReqId,Bin,T,awaiting_reset_ack,_,_)),
    retractall(request(ReqId,Bin,_)),
    retractall(pending_dup_notified(Bin)),
    clear_tried_listA(Bin),
    telemetryA(info,request_closed,ReqId,Bin,T,reset_acked).

reset_ackE(Bin,ReqId) :>
    \+ inflight(ReqId,Bin,_,awaiting_reset_ack,_,_),
    telemetryA(info,late_reset_ack,ReqId,Bin,none,ignored_after_async_close).

process_timeoutsA :-
    findall(inflight(ReqId,Bin,T,Stage,Ttl,R), inflight(ReqId,Bin,T,Stage,Ttl,R), L),
    forall(member(inflight(ReqId,Bin,T,Stage,Ttl,R), L),
        timeout_stepA(ReqId,Bin,T,Stage,Ttl,R)).

timeout_stepA(ReqId,Bin,T,Stage,Ttl,R) :-
    Ttl > 1,
    Ttl1 is Ttl - 1,
    retract(inflight(ReqId,Bin,T,Stage,Ttl,R)),
    assert(inflight(ReqId,Bin,T,Stage,Ttl1,R)).

timeout_stepA(ReqId,Bin,T,awaiting_reply,1,_) :-
    retractall(inflight(ReqId,Bin,T,awaiting_reply,1,_)),
    retractall(request(ReqId,Bin,T)),
    assert(tried(Bin,T)),
    telemetryA(warn,reply_timeout,ReqId,Bin,T,retry_select),
    select_truckI(ReqId,Bin).

timeout_stepA(ReqId,Bin,T,awaiting_assign_ack,1,R) :-
    R < 1,
    retractall(inflight(ReqId,Bin,T,awaiting_assign_ack,1,R)),
    assign_ack_ttl(Ttl),
    R1 is R + 1,
    assert(inflight(ReqId,Bin,T,awaiting_assign_ack,Ttl,R1)),
    shared_token(Token),
    messageA(T, send_message(assignment(Bin,ReqId,Token),control_center), control_center),
    messageA(T, send_message(assignment(Bin,ReqId),control_center), control_center),
    telemetryA(warn,assign_ack_timeout_resend,ReqId,Bin,T,resend_assignment).

timeout_stepA(ReqId,Bin,T,awaiting_assign_ack,1,R) :-
    R >= 1,
    retractall(inflight(ReqId,Bin,T,awaiting_assign_ack,1,R)),
    retractall(request(ReqId,Bin,T)),
    assert(tried(Bin,T)),
    telemetryA(error,assign_ack_timeout_drop,ReqId,Bin,T,retry_select),
    select_truckI(ReqId,Bin).

timeout_stepA(ReqId,Bin,T,awaiting_completion,1,_) :-
    retractall(inflight(ReqId,Bin,T,awaiting_completion,1,_)),
    retractall(request(ReqId,Bin,T)),
    assert(tried(Bin,T)),
    telemetryA(error,completion_timeout,ReqId,Bin,T,retry_select),
    select_truckI(ReqId,Bin).

timeout_stepA(ReqId,Bin,T,awaiting_reset_ack,1,_) :-
    retractall(inflight(ReqId,Bin,T,awaiting_reset_ack,1,_)),
    retractall(request(ReqId,Bin,_)),
    retractall(pending_dup_notified(Bin)),
    clear_tried_listA(Bin),
    telemetryA(warn,reset_ack_timeout,ReqId,Bin,T,closed_without_ack).

timeout_stepA(ReqId,Bin,none,awaiting_retry,1,_) :-
    retractall(inflight(ReqId,Bin,none,awaiting_retry,1,_)),
    clear_tried_listA(Bin),
    telemetryA(info,retry_dispatch,ReqId,Bin,none,reselect),
    select_truckI(ReqId,Bin).

clear_tried_listA(Bin) :-
    retractall(tried(Bin,_)).

monitorI(dummy) :>
    \+ started,
    evi(start).

monitorI(dummy) :>
    started,
    statistics(walltime,[Now,_]),
    last_cycle_ts(Last),
    cycle_interval_ms(Interval),
    Now - Last < Interval,
    !,
    true.

monitorI(dummy) :>
    started,
    statistics(walltime,[Now,_]),
    last_cycle_ts(Last),
    cycle_interval_ms(Interval),
    Now - Last >= Interval,
    retract(last_cycle_ts(Last)),
    assert(last_cycle_ts(Now)),
    evi(cycle).

monitor(dummy) :-
    evi(monitor(dummy)).
